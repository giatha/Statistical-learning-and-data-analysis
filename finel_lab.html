Q2.b Implemention Random Forest

The function takes as input an x matrix of explanatory variables of \(n*p\) size, vector y of the variable explained size \(n*1\) and number of trees. The function builds as many trees as it has received and puts them into a list. It then inserts the predicted values into a matrix so that each column is an expected value vector and then it averages over each row. The function returns as a list output that contains the average predicted values of all trees and a list of all trees.

my_rf <- function(x,y,num_tree,my_cp,maxd,mins){
  lst <- list()
  p_vec <- list()
  for (i in 1:num_tree){
    p <- sample(c(1:ncol(x)),size = sqrt(ncol(x)))
    p_vec[[i]] <- p
    sam <- sample(c(1:nrow(x)),nrow(x),replace = T)
    tre <- rpart(formula = y ~.,data = x[sam,p] , method  = "anova" , control = list(cp =  my_cp ,minsplit = mins, maxdepth = maxd))
    lst[[i]] <- tre
  }
  mat_tree <- matrix(nrow = nrow(x), ncol = num_tree)
  
  for (j in 1:length(lst)){
    mat_tree[,j] <- predict(lst[[j]],newdata = x)
  }
  y_hat1 <- apply(mat_tree, 1, mean)
  return(list(y_hat = y_hat1,model = lst,mat_tree,p = p_vec))
}
predict_my_rf <- function(rfmodel,x,p){
  mat_pre <- matrix( nrow = nrow(x), ncol = length(rfmodel))
  for (i in 1:length(rfmodel)) {
    mat_pre[,i] <- predict(rfmodel[[i]], newdata = x[,p[[i]]])
  }
  y_pre <- apply(mat_pre,1,mean)
  return(y_pre)
}

In this question I will use the model \(Gradient Boosting\) to predict the root of the number of article shares

my_prediction_model <- function(valid_x ,x_traini,y_traini,x_testi,y_testi){
  xgb_trcontrol = trainControl(
    method = "cv",
    number = 5,  
    allowParallel = TRUE,
    verboseIter = FALSE,
    returnData = FALSE
  )

  xgbGrid <- expand.grid(nrounds = c(100,200),  
                         max_depth = c(10, 15, 20, 25),
                         colsample_bytree = seq(0.5, 0.9, length.out = 5),
                         eta = 0.1,
                         gamma=0,
                         min_child_weight = 1,
                         subsample = 1
                        )

  xgb_model = train(
    x_traini, y_traini,  
    trControl = xgb_trcontrol,
    tuneGrid = xgbGrid,
    method = "xgbTree"
  )

  pred = predict(xgb_model, x_testi)
  Rmse = sqrt((sum(y_testi - pred)^2)/length(y_testi))
  sst = sum((y_testi - mean(y_testi))^2)
  sse = sum((y_testi - pred)^2)
  acc = sse/sst
  std_mod = acc*(1-acc)
  my_id = 302900683
  prd_valid <- predict(xgb_model, validation_X[,-c(1,2,3)])
  lst_model <- list("my_id" = my_id,"predicted_accuracy" = Rmse,"estimated_stderr" = std_mod,"prediction" = prd_valid)
  save(lst_model,file = "model.RData")
  return(lst_model)
}
